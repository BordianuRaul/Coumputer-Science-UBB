(defun elim (l current-level)
  (cond
    ((null l) nil)
    ((atom l) (list (list l current-level)))
    ((listp l)
     (mapcan (lambda (x) (elim x (+ current-level 1))) l))
    (t nil)))

(defun extract-first-elements (nested-list)
  (mapcar #'car nested-list))

(print (extract-first-elements '((A 1) (B 2) (C 2) (D 3) (E 3))))
